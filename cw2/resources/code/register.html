//task 4
int ran = 0;
int falsePasses = 0;
int loopCount = 0;

//function for visiting leaves
void leaf(Node *node, double tolerence, int choice){
  for(int i = 0; i < 4; i++){
    if(node->child[i] != NULL){
      if(indicator(node->child[i], tolerence, choice) == false){
        for(int j = 0; j < 4; j++){
          printf("child++\n");
          makeChildren(node->child[j], 50);
        }
      }
      else{
        falsePasses++;
      }
      leaf(node->child[i], tolerence, choice);
    }
  }
}

//function for looping
void task4(Node *node, double tolerence, int choice){
  if(ran == 0){
    leaf(node, tolerence, choice);
  }
  else if(falsePasses > 0){
    ran++;
    falsePasses = 0;
    leaf(node, tolerence, choice);
  }
  else{
    destroyTree(node);
  }
}
